VBA Code Examples

Copy Data


Sub CopyData()
    Sheets("Sheet1").Range("A1:A10").Copy
    Sheets("Sheet2").Range("B1").PasteSpecial Paste:=xlPasteValues
End Sub

Sum Range


Sub SumRange()
    Dim total As Double
    total = Application.WorksheetFunction.Sum(Range("A1:A10"))
    MsgBox "The total is " & total
End Sub
